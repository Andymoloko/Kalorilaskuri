    <script>
        function calculate() {
            const weight = parseFloat(document.getElementById("weight").value);
            const height = parseFloat(document.getElementById("height").value);
            const age = parseInt(document.getElementById("age").value);
            const gender = document.getElementById("gender").value;
            const activity = parseFloat(document.getElementById("activity").value);

            if (!weight || !height || !age || !activity) {
                document.getElementById("results").innerHTML = "<p style='color:red;'>Täytä kaikki kentät!</p>";
                return;
            }

            let bmr = gender === "male"
                ? 10 * weight + 6.25 * height - 5 * age + 5
                : 10 * weight + 6.25 * height - 5 * age - 161;

            let tdee = bmr * activity;

            const calorieDeficit = (7700 * 2) / 30;
            let weightLossCalories = tdee - calorieDeficit;

            document.getElementById("results").innerHTML = `
                <strong>BMR:</strong> ${bmr.toFixed(2)} kcal<br>
                <strong>TDEE:</strong> ${tdee.toFixed(2)} kcal<br>
                <strong>Painonpudotus:</strong> ${weightLossCalories.toFixed(2)} kcal/päivä
            `;
        }
    </script>
